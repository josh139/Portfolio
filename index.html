 CREATE TABLE Supplier
(
	supID		INTEGER(10) NOT NULL,
	supProd		VARCHAR(20) NOT NULL,
	supName		VARCHAR(15),
	supAddress	VARCHAR(50) NOT NULL,
	supPostCode VARCHAR(12) NOT NULL,
	CONSTRAINT s_spid_sppd_pk PRIMARY KEY (supID, supProd)
);

CREATE TABLE Customer
(
	custID			INTEGER(10),
	custFName		VARCHAR(10) NOT NULL,
	custSName		VARCHAR(10) NOT NULL,
	custAddress		VARCHAR(50) NOT NULL,
	custPostCode	VARCHAR(12) NOT NULL,	
	custEmail		VARCHAR(30) NOT NULL,
	custTelNo		BIGINT(11) NOT NULL,
	CONSTRAINT c_cid_pk PRIMARY KEY (custID)
);

CREATE TABLE Customer_Order
(
	orderID			INTEGER(10) NOT NULL,
	custID			INTEGER(10),
	orderBuyDate	DATE,
	orderDelDate	DATE,
	CONSTRAINT c_oid_pk PRIMARY KEY (orderID),
	CONSTRAINT c_cid_fk FOREIGN KEY (custID)
	REFERENCES Customer (custID)
);

CREATE TABLE Product
(
	prodNo 			INTEGER(10) NOT NULL,
	supID			INTEGER(10) NOT NULL,
	supProd			VARCHAR(20)NOT NULL,
	prodQuantity	INTEGER(5) NOT NULL,
	prodStock		INTEGER(10) NOT NULL,
	CONSTRAINT p_pno_pk PRIMARY KEY (prodNo),
	CONSTRAINT p_spid_sppd_fk FOREIGN KEY (supID, supProd)
	REFERENCES Supplier (supID, supProd)
);

CREATE TABLE Set_Type
(
	setTypeID	INTEGER(10) NOT NULL,
	prodNo1		INTEGER(10) NOT NULL,
	prodNo2		INTEGER(10) NOT NULL,
	prodNo3		INTEGER(10) NOT NULL,
	prodNo4		INTEGER(10),
	prodNo5		INTEGER(10),
	setGender	CHAR(1) NOT NULL,
	CONSTRAINT st_styid_pk PRIMARY KEY (setTypeID),
	CONSTRAINT st_pno1_fk FOREIGN KEY (prodNo1)
	REFERENCES Product (prodNo),
	CONSTRAINT st_pno2_fk FOREIGN KEY (prodNo2)
	REFERENCES Product (prodNo),
	CONSTRAINT st_pno3_fk FOREIGN KEY (prodNo3)
	REFERENCES Product (prodNo),
	CONSTRAINT st_pno4_fk FOREIGN KEY (prodNo4)
	REFERENCES Product (prodNo),
	CONSTRAINT st_pno5_fk FOREIGN KEY (prodNo5)
	REFERENCES Product (prodNo)
);

CREATE TABLE Finished_Set
(
	setID		INTEGER(10) NOT NULL,
	setName		VARCHAR(20) NOT NULL,
	setPrice	DECIMAL(5,2) NOT NULL,
	setTypeID	INTEGER(10) NOT NULL,
	setStock	INTEGER(10) NOT NULL,
	CONSTRAINT fs_stid_stnm_stpr_pk PRIMARY KEY (setID,setName,setPrice),
	CONSTRAINT fs_styid_fk FOREIGN KEY (setTypeID)
	REFERENCES Set_Type (setTypeID)
);

CREATE TABLE Set_Order
(
	orderTotal	DECIMAL(5,2) NOT NULL,
	orderID		INTEGER(10) NOT NULL,
	setID		INTEGER(10) NOT NULL,
	setName		VARCHAR(20) NOT NULL,
	setPrice	DECIMAL(5,2) NOT NULL,
	numSets		INTEGER(3),
	CONSTRAINT s_ot_pk PRIMARY KEY (orderTotal),	
	CONSTRAINT s_oid_fk FOREIGN KEY (orderID) 	
	REFERENCES Customer_Order (orderID),
	CONSTRAINT fs_stid_stnm_stpr_fk FOREIGN KEY (setID,setName,setPrice)
	REFERENCES Finished_Set (setID,setName,setPrice)
);
